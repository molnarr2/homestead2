# Goal

Define the Firestore data model (collections/subcollections) and write security rules. All farm data lives under `/homestead/{homesteadId}/` as subcollections — a homestead is the central organizing entity. Users are members of homesteads, enabling multi-user access (invite a caretaker, share a farm). User profiles and feedback remain at root level. Rules live in `/firebase-rule`.

**Important:** We share one Firebase project with the existing production app. The `user` collection is reused (with a `schemaVersion` field to distinguish v2 users), and all other collections use a `_v2` suffix to avoid collision with existing production data. The `homestead` collection is new and does not need a suffix.

## 2. Schema.ts Mapping (Firebase library)

```typescript
export function schemaToDocPaths(schema: SchemaType): string[] {
  switch (schema) {
    // Root-level (user-scoped)
    case SchemaType.User:                    return ["user"]
    case SchemaType.UserDevice:              return ["user", "device"]
    case SchemaType.Feedback:                return ["feedback_v2"]

    // Root-level (homestead)
    case SchemaType.Homestead:               return ["homestead"]
    case SchemaType.HomesteadMember:         return ["homestead", "member"]

    // Homestead-scoped subcollections
    case SchemaType.Animal:                  return ["homestead", "animal_v2"]
    case SchemaType.CareEvent:               return ["homestead", "careEvent_v2"]
    case SchemaType.HealthRecord:            return ["homestead", "healthRecord_v2"]
    case SchemaType.BreedingRecord:          return ["homestead", "breedingRecord_v2"]
    case SchemaType.Note:                    return ["homestead", "note_v2"]
    case SchemaType.WeightLog:               return ["homestead", "weightLog_v2"]
    case SchemaType.ProductionLog:           return ["homestead", "productionLog_v2"]
    case SchemaType.AnimalType:              return ["homestead", "animalType_v2"]
    case SchemaType.AnimalTypeBreed:         return ["homestead", "animalType_v2", "breed"]
    case SchemaType.AnimalTypeCareTemplate:  return ["homestead", "animalType_v2", "careTemplate"]
    case SchemaType.AnimalTypeEventTemplate: return ["homestead", "animalType_v2", "eventTemplate"]
    default: throw new Error(`Unknown schema type: ${schema}`)
  }
}
```

## 3. Firestore Security Rules (`firebase-rule/firestore.rules`)

```
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    function isAuth() {
      return request.auth != null;
    }

    function isOwner(userId) {
      return request.auth.uid == userId;
    }

    // Check if the authenticated user is a member of the homestead
    function isMember(homesteadId) {
      return isAuth() && exists(/databases/$(database)/documents/homestead/$(homesteadId)/member/$(request.auth.uid));
    }

    // Check if the authenticated user has a specific role (or higher) in the homestead
    // Role hierarchy: owner > manager > caretaker > viewer
    function getMemberRole(homesteadId) {
      return get(/databases/$(database)/documents/homestead/$(homesteadId)/member/$(request.auth.uid)).data.role;
    }

    function isHomesteadOwner(homesteadId) {
      return isMember(homesteadId) && getMemberRole(homesteadId) == 'owner';
    }

    function canWrite(homesteadId) {
      return isMember(homesteadId) && getMemberRole(homesteadId) in ['owner', 'manager', 'caretaker'];
    }

    function canManage(homesteadId) {
      return isMember(homesteadId) && getMemberRole(homesteadId) in ['owner', 'manager'];
    }

    // ──────────────────────────────────────────────
    // Root-level: User profile — owner only
    // ──────────────────────────────────────────────
    match /user/{userId} {
      allow read: if isAuth() && isOwner(userId);
      allow create: if isAuth() && isOwner(userId);
      allow update: if isAuth() && isOwner(userId);
      allow delete: if false;

      match /device/{deviceId} {
        allow read, write: if isAuth() && isOwner(userId);
      }
    }

    // ──────────────────────────────────────────────
    // Root-level: Homestead — members can read, owner manages
    // ──────────────────────────────────────────────
    match /homestead/{homesteadId} {
      allow read: if isMember(homesteadId);
      allow create: if isAuth();  // any authenticated user can create a homestead
      allow update: if isHomesteadOwner(homesteadId);
      allow delete: if false;

      // Members subcollection — owner/manager can manage members
      match /member/{memberId} {
        allow read: if isMember(homesteadId);
        allow create: if canManage(homesteadId);
        allow update: if canManage(homesteadId);
        allow delete: if isHomesteadOwner(homesteadId);
      }

      // ──────────────────────────────────────────
      // Homestead-scoped data — members read, writers write
      // ──────────────────────────────────────────

      // Animals
      match /animal_v2/{animalId} {
        allow read: if isMember(homesteadId);
        allow create: if canWrite(homesteadId);
        allow update: if canWrite(homesteadId);
        allow delete: if false;
      }

      // Care Events
      match /careEvent_v2/{eventId} {
        allow read: if isMember(homesteadId);
        allow create: if canWrite(homesteadId);
        allow update: if canWrite(homesteadId);
        allow delete: if false;
      }

      // Health Records
      match /healthRecord_v2/{recordId} {
        allow read: if isMember(homesteadId);
        allow create: if canWrite(homesteadId);
        allow update: if canWrite(homesteadId);
        allow delete: if false;
      }

      // Breeding Records
      match /breedingRecord_v2/{recordId} {
        allow read: if isMember(homesteadId);
        allow create: if canWrite(homesteadId);
        allow update: if canWrite(homesteadId);
        allow delete: if false;
      }

      // Notes
      match /note_v2/{noteId} {
        allow read: if isMember(homesteadId);
        allow create: if canWrite(homesteadId);
        allow update: if canWrite(homesteadId);
        allow delete: if false;
      }

      // Weight Logs
      match /weightLog_v2/{logId} {
        allow read: if isMember(homesteadId);
        allow create: if canWrite(homesteadId);
        allow update: if canWrite(homesteadId);
        allow delete: if false;
      }

      // Production Logs
      match /productionLog_v2/{logId} {
        allow read: if isMember(homesteadId);
        allow create: if canWrite(homesteadId);
        allow update: if canWrite(homesteadId);
        allow delete: if false;
      }

      // Animal Types — manager+ can modify (structural data)
      match /animalType_v2/{typeId} {
        allow read: if isMember(homesteadId);
        allow create: if canManage(homesteadId);
        allow update: if canManage(homesteadId);
        allow delete: if false;

        match /breed/{breedId} {
          allow read: if isMember(homesteadId);
          allow create: if canManage(homesteadId);
          allow update: if canManage(homesteadId);
          allow delete: if false;
        }

        match /careTemplate/{templateId} {
          allow read: if isMember(homesteadId);
          allow create: if canManage(homesteadId);
          allow update: if canManage(homesteadId);
          allow delete: if false;
        }

        match /eventTemplate/{templateId} {
          allow read: if isMember(homesteadId);
          allow create: if canManage(homesteadId);
          allow update: if canManage(homesteadId);
          allow delete: if false;
        }
      }
    }

    // ──────────────────────────────────────────────
    // Root-level: Feedback — write-only, read by admin
    // ──────────────────────────────────────────────
    match /feedback_v2/{feedbackId} {
      allow create: if isAuth();
      allow read: if false;
      allow update: if false;
      allow delete: if false;
    }
  }
}
```

**Security rule cost note:** `isMember()` and role checks each cost one document read per rule evaluation. For the MVP single-user case, every request will hit the member doc — this is a single small read and is acceptable. For high-traffic scenarios, consider caching membership in custom auth claims via Cloud Functions.

## 4. Storage Rules (`firebase-rule/storage.rules`)

```
rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    // User avatars — owner writes, authenticated reads
    match /users/{userId}/{allPaths=**} {
      allow read: if request.auth != null;
      allow write: if request.auth.uid == userId;
    }
    // Homestead data (animal photos, etc.) — authenticated members
    // Note: Storage rules cannot check Firestore membership, so we rely on
    // authenticated access + the app only providing paths for valid homesteads.
    // For stricter enforcement, use Cloud Functions for uploads.
    match /homesteads/{homesteadId}/{allPaths=**} {
      allow read: if request.auth != null;
      allow write: if request.auth != null;
    }
  }
}
```

# Common Query Patterns

All queries are scoped to a homestead via the collection path — no `userId` filter needed.

```javascript
// Reference to the homestead's subcollection
const hsRef = db.collection('homestead').doc(homesteadId)

// All overdue + due-today care across all animals
hsRef.collection('careEvent_v2')
  .where('admin.deleted', '==', false)
  .where('completedDate', '==', '')
  .where('dueDate', '<=', today)
  .orderBy('dueDate')

// All active withdrawals across all animals
hsRef.collection('healthRecord_v2')
  .where('admin.deleted', '==', false)
  .where('withdrawalPeriodDays', '>', 0)

// All active breedings across all animals
hsRef.collection('breedingRecord_v2')
  .where('admin.deleted', '==', false)
  .where('status', '==', 'active')
  .orderBy('expectedDueDate')

// Animals by type
hsRef.collection('animal_v2')
  .where('admin.deleted', '==', false)
  .where('animalType', '==', typeName)
  .orderBy('name')
```

# Notes

- **Homestead-scoped architecture:** All farm data lives under `/homestead/{homesteadId}/` as subcollections. Queries are scoped by the collection path — no `userId` field or filter needed on farm data.
- **Multi-user access built in:** Users are members of homesteads with roles (`owner`, `manager`, `caretaker`, `viewer`). Security rules check membership via the `member` subcollection.
- **Role-based permissions:** `viewer` = read only, `caretaker` = read + write farm data, `manager` = caretaker + manage animal types/members, `owner` = full control including homestead settings and member removal.
- **Membership cost:** Each security rule evaluation costs one doc read to check membership (and one more for role checks). This is acceptable — the member doc is small and Firestore caches within a request.
- **User ↔ Homestead mapping:** `User.homesteadIds[]` allows quick lookup of a user's homesteads without querying all homestead member subcollections. Keep this in sync when adding/removing members.
- No hard deletes — all documents use soft-delete via `admin.deleted`
- `animalType` subcollection rules (`breed`, `careTemplate`, `eventTemplate`) inherit from the parent homestead membership check — no separate ownership field needed
- Storage paths use `/homesteads/{homesteadId}/` prefix for farm data and `/users/{userId}/` for user avatars
