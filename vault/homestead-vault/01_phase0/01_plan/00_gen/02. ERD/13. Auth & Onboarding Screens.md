# Goal

Build the authentication and onboarding feature screens. This is the first user-facing flow: Login, Register, Species Selection, and Loading. Follows the Controller + Screen separation pattern. After registration, users select which species they keep — this seeds starter animal types, breeds, and care templates.

## 3. Login Screen

### `LoginController.ts`
```typescript
export function useLoginController(navigation) {
  const [email, setEmail] = useState('')
  const [password, setPassword] = useState('')
  const [loading, setLoading] = useState(false)
  const [error, setError] = useState('')

  const login = async () => {
    setLoading(true)
    setError('')
    const result = await bsAuthService.signin(email, password)
    if (result.error) {
      setError(result.error)
    }
    setLoading(false)
  }

  const goToRegister = () => navigation.navigate('Register')
  const forgotPassword = async () => {
    if (!email) { setError('Enter your email first'); return }
    await bsAuthService.sendPasswordResetEmail(email)
  }

  return { email, setEmail, password, setPassword, loading, error, login, goToRegister, forgotPassword }
}
```

## 4. Register Screen

### `RegisterController.ts`
```typescript
export function useRegisterController(navigation) {
  const [firstName, setFirstName] = useState('')
  const [lastName, setLastName] = useState('')
  const [email, setEmail] = useState('')
  const [password, setPassword] = useState('')
  const [loading, setLoading] = useState(false)
  const [error, setError] = useState('')

  const register = async () => {
    setLoading(true)
    setError('')
    const result = await bsAuthService.createAccount(email, password)
    if (result.error) {
      setError(result.error)
      setLoading(false)
      return
    }
    // Create user document in Firestore
    const userId = bsAuthService.getUserId()
    await bsAuthFeatureService.createUserProfile(userId, firstName, lastName, email)
    // Navigate to species selection
    navigation.navigate('SpeciesSelection')
    setLoading(false)
  }

  return { firstName, setFirstName, lastName, setLastName, email, setEmail, password, setPassword, loading, error, register }
}
```

### `RegisterScreen.tsx`
- Fields: First Name, Last Name, Email, Password
- Register button
- "Already have an account? Login" link
- Error display
- Validation: email format, password minimum 6 chars, first name required

```typescript
export function useSpeciesSelectionController(navigation) {
  const [selectedSpecies, setSelectedSpecies] = useState<string[]>([])
  const [loading, setLoading] = useState(false)

  const toggleSpecies = (species: string) => {
    setSelectedSpecies(prev =>
      prev.includes(species)
        ? prev.filter(s => s !== species)
        : [...prev, species]
    )
  }

  const complete = async () => {
    if (selectedSpecies.length === 0) return
    setLoading(true)
    // Save selected species to user document
    await useUserStore.getState().updateUser({ selectedSpecies, onboardingComplete: true })
    // Seed starter playbooks (animal types, breeds, care templates)
    await useAnimalTypeStore.getState().seedStarterPlaybooks(selectedSpecies)
    setLoading(false)
    // Auth state change drives nav to main app
  }

  return { availableSpecies: AVAILABLE_SPECIES, selectedSpecies, toggleSpecies, loading, complete }
}
```

# Notes

- Auth state changes automatically trigger navigation via Zustand authStore → RootNavigation
- No need to manually navigate after login — Firebase auth state listener handles it
- Species selection happens once after registration — seeds starter data for the customization engine
- Starter playbooks include common breeds and care templates per species (e.g., Goat → Alpine, Nubian breeds + deworming schedule, hoof trimming)
- Keep the UI warm and welcoming — this is the first impression
- Validation: email format, password minimum 6 chars, first name required
