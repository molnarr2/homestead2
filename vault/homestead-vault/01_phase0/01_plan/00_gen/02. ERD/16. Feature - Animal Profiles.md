# Goal

Build the Animals tab with animal list grouped by type, search/filter, animal CRUD, animal detail screen with all sub-sections (health records, breeding, care history, notes, weight logs), photo upload, and state management (own/sold/died/processed). The Animal Detail screen is the hub — all per-animal data is accessed from here.

## Animal List Screen (Animals Tab)

### `AnimalListController.ts`
```typescript
export function useAnimalListController(navigation) {
  const { animals, fetchAnimals, loading } = useAnimalStore()
  const [searchQuery, setSearchQuery] = useState('')
  const [filterState, setFilterState] = useState<AnimalState | 'all'>('all')

  useFocusEffect(useCallback(() => {
    fetchAnimals()
  }, []))

  const filteredAnimals = animals.filter(a => {
    const matchesSearch = a.name.toLowerCase().includes(searchQuery.toLowerCase()) ||
      a.breed.toLowerCase().includes(searchQuery.toLowerCase())
    const matchesState = filterState === 'all' || a.state === filterState
    return matchesSearch && matchesState
  })

  // Group by animalType
  const groupedAnimals = groupBy(filteredAnimals, 'animalType')

  const onAnimalPress = (animalId: string) => navigation.navigate('AnimalDetail', { animalId })
  const onCreateAnimal = () => navigation.navigate('CreateAnimal')

  return { groupedAnimals, searchQuery, setSearchQuery, filterState, setFilterState, loading, onAnimalPress, onCreateAnimal }
}
```

export function useAnimalDetailController(navigation, route) {
  const { animalId } = route.params
  const { animals } = useAnimalStore()
  const animal = animals.find(a => a.id === animalId)

  // Fetch related data
  const { fetchCareEventsByAnimal, careEvents } = useCareStore()
  const { fetchHealthRecordsByAnimal, healthRecords } = useHealthStore()
  const { fetchBreedingRecordsByAnimal, breedingRecords } = useBreedingStore()
  const { fetchNotesByAnimal, notes } = useNoteStore()
  const { fetchWeightLogsByAnimal, weightLogs } = useWeightStore()

  const [activeTab, setActiveTab] = useState('timeline')

  useEffect(() => {
    fetchCareEventsByAnimal(animalId)
    fetchHealthRecordsByAnimal(animalId)
    fetchBreedingRecordsByAnimal(animalId)
    fetchNotesByAnimal(animalId)
    fetchWeightLogsByAnimal(animalId)
  }, [animalId])

  // Calculate derived data
  const age = calculateAnimalAge(animal?.birthday ?? '')
  const activeWithdrawals = getActiveWithdrawals(healthRecords)

  const onEdit = () => navigation.navigate('EditAnimal', { animalId })
  const onAddCare = () => navigation.navigate('CreateCareEvent', { animalId })
  const onAddHealth = () => navigation.navigate('CreateHealthRecord', { animalId })
  const onAddBreeding = () => navigation.navigate('CreateBreedingRecord', { animalId })
  const onAddNote = () => navigation.navigate('CreateNote', { animalId })
  const onAddWeight = () => navigation.navigate('CreateWeightLog', { animalId })
  const onAddProduction = () => navigation.navigate('CreateProductionLog', { animalId })

  return {
    animal, age, activeWithdrawals, activeTab, setActiveTab,
    careEvents, healthRecords, breedingRecords, notes, weightLogs,
    onEdit, onAddCare, onAddHealth, onAddBreeding, onAddNote, onAddWeight, onAddProduction,
  }
}

## Create Animal Screen

### `CreateAnimalController.ts`
```typescript
export function useCreateAnimalController(navigation, route) {
  const { createAnimal } = useAnimalStore()
  const { animalTypes } = useAnimalTypeStore()
  const userId = useAuthStore().userId

  const [name, setName] = useState('')
  const [animalTypeId, setAnimalTypeId] = useState(route.params?.animalTypeId ?? '')
  const [breedId, setBreedId] = useState('')
  const [birthday, setBirthday] = useState('')
  const [gender, setGender] = useState<AnimalGender>('unknown')
  const [color, setColor] = useState('')
  const [photoUri, setPhotoUri] = useState('')
  const [sireId, setSireId] = useState('')
  const [damId, setDamId] = useState('')
  const [loading, setLoading] = useState(false)

  const submit = async () => {
    if (!name.trim()) return
    setLoading(true)
    await createAnimal({
      userId, name, animalTypeId, breedId, birthday, gender, color,
      photoUri, sireId, damId, state: 'own',
    })
    setLoading(false)
    navigation.goBack()
  }

  return { /* all state and setters */ submit, animalTypes, loading }
}
```

## Animal Service (`src/feature/animal/service/`)

### `IAnimalService.ts`
```typescript
export default interface IAnimalService {
  fetchAnimalsByUser(userId: string): Promise<Animal[]>
  createAnimal(animal: Animal, photoUri?: string): Promise<IResult>
  updateAnimal(animal: Animal, photoUri?: string): Promise<IResult>
  softDeleteAnimal(animalId: string): Promise<IResult>
  updateAnimalState(animalId: string, state: AnimalState): Promise<IResult>
}
```

### `AnimalService.ts`
- Uses `ICloudPlatform` for Firestore operations
- `fetchAnimalsByUser`: query `animal` collection where `userId == uid` AND `admin.deleted == false`
- `createAnimal`: `setDoc` with auto-generated ID, upload photo to Storage if provided
- `updateAnimalState`: updates only the `state` field
- Photo upload: store in `animals/{animalId}/photo.jpg`, save `photoStorageRef` and `photoUrl`
- Registered in `Bootstrap.ts`

# Subscription Gating

- **Free tier**: Maximum 10 animals. On create, check count. If at limit, show Pro paywall.
- **Pro/Farm tier**: Unlimited animals.

# Notes

- Animal Detail is the hub — the most important screen in the app
- All per-animal data (health, breeding, care, notes, weight) is accessed from tabs within Animal Detail
- State transitions: `own` → `sold` / `died` / `processed` (one-way, set via edit screen)
- Parent linking: `sireId` and `damId` reference other animal documents — enables lineage tracking
- Age calculated client-side from `birthday` using `calculateAnimalAge()` utility
- Photo upload uses Firebase Storage → download URL stored on the animal document
- Search is client-side (all animals loaded into Zustand store). For very large herds (100+), consider pagination.
