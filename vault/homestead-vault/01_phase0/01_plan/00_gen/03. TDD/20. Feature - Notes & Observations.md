# Goal

Build the notes system with quick timestamped entries per animal, tag-based categorization, optional photo attachment, and display on the Animal Detail timeline. Notes are the simplest per-animal data type — they capture the observations and context that don't fit neatly into structured records.

**Architecture notes:**
- `INoteService.ts` (interface) + `NoteService.ts` (implementation) — Interface + Implementation pattern
- `NoteService` calls `@react-native-firebase/firestore` directly via homestead subcollection: `this.homesteadRef.collection('note')`
- Photo upload uses `@react-native-firebase/storage` directly inside `NoteService`
- AdminObject uses Firebase Timestamps (`Tstamp` from `apps/mobile/src/schema/type/Tstamp.ts`)
- Types (`Note`, `NoteTag`) imported from `schema/notes/Note.ts` — not re-defined in the service (D9)
- No `userId` in queries or documents — homestead scoping replaces user scoping
- Service exported from `Bootstrap.ts` as `bsNoteService`

## Firestore Schema

**Collection:** `/homestead/{homesteadId}/note/{noteId}` (homestead subcollection)

| Field | Type | Required | Description |
|-------|------|----------|-------------|
| `id` | `string` | Yes | Document ID |
| `admin` | `AdminObject` | Yes | `deleted`, `updated_at` (Tstamp), `created_at` (Tstamp) |
| `animalId` | `string` | Yes | Reference to the animal this note belongs to |
| `tags` | `string[]` | Yes | Categorization tags: `"Health"`, `"Behavior"`, `"Breeding"`, `"Feed"`, `"Production"`, `"General"` |
| `text` | `string` | Yes | Note content (multiline) |
| `photoStorageRef` | `string` | No | Firebase Storage path |
| `photoUrl` | `string` | No | Public download URL |

## Create Note Screen

### `CreateNoteController.ts`
```typescript
import { useState } from 'react'
import { NoteTag } from '../../../schema/notes/Note'
import { bsNoteService } from '../../../Bootstrap'

export function useCreateNoteController(navigation: any, route: any) {
  const { animalId } = route.params

  const [text, setText] = useState('')
  const [tags, setTags] = useState<NoteTag[]>([])
  const [photoUri, setPhotoUri] = useState('')
  const [loading, setLoading] = useState(false)

  const toggleTag = (tag: NoteTag) => {
    setTags(prev =>
      prev.includes(tag) ? prev.filter(t => t !== tag) : [...prev, tag]
    )
  }

  const submit = async () => {
    if (!text.trim()) return
    setLoading(true)
    await bsNoteService.createNote(
      { id: '', admin: {} as any, animalId, text, tags },
      photoUri || undefined,
    )
    setLoading(false)
    navigation.goBack()
  }

  return { text, setText, tags, toggleTag, photoUri, setPhotoUri, loading, submit }
}
```

## Note Detail Screen

### `NoteDetailController.ts`
```typescript
import { useState, useEffect } from 'react'
import { bsNoteService } from '../../../Bootstrap'
import { Note } from '../../../schema/notes/Note'

export function useNoteDetailController(navigation: any, route: any) {
  const { noteId, animalId } = route.params
  const [note, setNote] = useState<Note | null>(null)

  useEffect(() => {
    loadNote()
  }, [noteId])

  const loadNote = async () => {
    const notes = await bsNoteService.fetchNotesByAnimal(animalId)
    const found = notes.find(n => n.id === noteId)
    if (found) setNote(found)
  }

  const onDelete = async () => {
    await bsNoteService.deleteNote(noteId)
    navigation.goBack()
  }

  return { note, onDelete }
}
```

### `NoteDetailScreen.tsx`

## Note Service

### Interface (`apps/mobile/src/feature/notes/service/INoteService.ts`)

```typescript
import { Note } from '../../../schema/notes/Note'
import { IResult } from '../../../util/Result'

export interface INoteService {
  fetchNotesByAnimal(animalId: string): Promise<Note[]>
  createNote(note: Note, photoUri?: string): Promise<IResult>
  deleteNote(noteId: string): Promise<IResult>
}
```

### Implementation (`apps/mobile/src/feature/notes/service/NoteService.ts`)

The service calls `@react-native-firebase/firestore` and `@react-native-firebase/storage` directly via homestead subcollection. Types are imported from schema — not re-defined here (D9).

```typescript
import firestore, { FirebaseFirestoreTypes } from '@react-native-firebase/firestore'
import storage from '@react-native-firebase/storage'
import { IResult, SuccessResult, ErrorResult } from '../../../util/Result'
import { adminObject_default } from '../../../schema/object/AdminObject'
import Log from '../../../library/log/Log'
import { Note } from '../../../schema/notes/Note'
import { INoteService } from './INoteService'

export default class NoteService implements INoteService {
  private homesteadRef: FirebaseFirestoreTypes.DocumentReference

  constructor(homesteadId: string) {
    this.homesteadRef = firestore().collection('homestead').doc(homesteadId)
  }

  async fetchNotesByAnimal(animalId: string): Promise<Note[]> {
    try {
      const snapshot = await this.homesteadRef
        .collection('note')
        .where('animalId', '==', animalId)
        .where('admin.deleted', '==', false)
        .orderBy('admin.created_at', 'desc')
        .get()

      return snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() } as Note))
    } catch (error) {
      Log.error('NoteService', 'fetchNotesByAnimal failed: ' + error)
      return []
    }
  }

  async createNote(note: Note, photoUri?: string): Promise<IResult> {
    try {
      const docRef = this.homesteadRef.collection('note').doc()
      const noteData: Note = {
        ...note,
        id: docRef.id,
        admin: adminObject_default(),
      }

      if (photoUri) {
        const storagePath = `homestead/${this.homesteadRef.id}/note/${docRef.id}/photo.jpg`
        await storage().ref(storagePath).putFile(photoUri)
        const downloadUrl = await storage().ref(storagePath).getDownloadURL()
        noteData.photoStorageRef = storagePath
        noteData.photoUrl = downloadUrl
      }

      await docRef.set(noteData)
      return SuccessResult
    } catch (error) {
      Log.error('NoteService', 'createNote failed: ' + error)
      return ErrorResult('Failed to create note: ' + error)
    }
  }

  async deleteNote(noteId: string): Promise<IResult> {
    try {
      await this.homesteadRef.collection('note').doc(noteId).update({
        'admin.deleted': true,
        'admin.updated_at': firestore.FieldValue.serverTimestamp(),
      })
      return SuccessResult
    } catch (error) {
      Log.error('NoteService', 'deleteNote failed: ' + error)
      return ErrorResult('Failed to delete note: ' + error)
    }
  }
}
```

# Notes

- Notes are the simplest per-animal record type — quick capture of observations
- They supplement structured records (health, breeding, care) with free-form context
- Tags enable filtering on the Animal Detail timeline
- Photo attachment is optional but useful for documenting visible symptoms, injuries, or condition
- Notes display in the Animal Detail timeline alongside other record types
- Soft-delete only — notes are never hard-deleted (`admin.deleted = true`)
- No `onSnapshot` needed — notes are fetched on demand when viewing an animal's detail
- All data scoped under `/homestead/{homesteadId}/note/` — no `userId` field needed
- Service implements `INoteService` interface — Interface + Implementation pattern
- Photo storage path: `homestead/{homesteadId}/note/{noteId}/photo.jpg`
- AdminObject timestamps use Firebase `Tstamp` (server timestamps on create, updated on mutation)
